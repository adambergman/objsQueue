objsQueue - Simple Example

This example will queue an objective-c function. Alerts will appear when the commands are finished.

Run a Command! runCommandNoParams()

Run another Command! runCommand()


JavaScript Source 
function runCommandNoParams()
{
    objsQueue.add("methodName");
}
            
function runCommand()
{
    var didComplete = function(obj){ alert('runCommand didComplete.  Objective-C returned: "' + obj.value + '"'); };
    var didFail = function(obj){ alert('runCommand didFail ' + obj.value); };
    objsQueue.add("methodName:secondString:", {params:['test','test2'], complete:didComplete, complete:didFail);
}
