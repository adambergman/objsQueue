objsQueue - Simple Example

This example will queue an objective-c function. Alerts will appear when the command is finished.

Run a Command!


JavaScript Source 
function runCommand()
{
    var didComplete = function(obj){ alert('runCommand didComplete.  Objective-C returned: "' + obj.value + '"'); };
    var didFail = function(obj){ alert('runCommand didFail ' + obj.value); };
    objsQueue.add("methodName:secondString:", ['test','test2'], didComplete, didFail);
}
